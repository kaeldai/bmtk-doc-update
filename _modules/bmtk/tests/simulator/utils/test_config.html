[docs]def test_resolve_var_list():
    """Check variables can be resolved in list"""
    config_file = {
        'manifest': {
            '$p1': 'a',
            '$p2': 'b'
        },
        'l1': ['$p1/test', '${p2}/test', 9]
    }
    conf = cfg.from_dict(config_file)
    assert(conf['l1'][0] == 'a/test')
    assert(conf['l1'][1] == 'b/test')
    assert(conf['l1'][2] == 9)




[docs]def test_resolve_var_dict():
    """Check variables can be resolved in dictionary"""
    config_file = {
        'manifest': {
            '$v1': 'a',
            '$v2': 'c'
        },
        'd1': {
            'k1': '$v1',
            'k2': 'B',
            'k3': ['${v2}'],
            'k4': 4
        }
    }
    conf = cfg.from_dict(config_file)
    assert(conf['d1']['k1'] == 'a')
    assert(conf['d1']['k2'] == 'B')
    assert(conf['d1']['k3'] == ['c'])
    assert(conf['d1']['k4'] == 4)
